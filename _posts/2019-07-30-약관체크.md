```javascript
$('#check-all-agree').click(function(){

    var chk = $(this).is(':checked');

    if(chk){

        $('.sign-up-agree--etc input[type="checkbox"]').prop('checked',true);
        $('.button--disable').addClass('is-active');

    }else{

        $('.sign-up-agree--etc input[type="checkbox"]').prop('checked',false);
        $('.button--disable').removeClass('is-active');

    }

});

$('input[name="must-agree"]').click(function(){

    var chk1 = $('#check-agree-01').is(':checked');
    var chk2 = $('#check-agree-02').is(':checked');
    var chk3 = $('#check-agree-03').is(':checked');

    if( chk1 == false || chk2 == false || chk3 == false ){

        $('#check-all-agree').prop('checked',false);
        $('.button--disable').removeClass('is-active');

    }else{

        $('.button--disable').addClass('is-active');

    }

});
```
